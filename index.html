// 데이터를 불러와서 화면에 반영하는 핵심 로직 fetch('AI_재정사업_표준화_533.json') .then(response => { if (!response.ok) throw new Error("파일을 찾을 수 없습니다."); return response.json(); }) .then(data => { const projects = data.projects; // 1. 2026년 본예산 총액 계산 let totalSum = 0; projects.forEach(p => { const budget2026 = p.예산.find(b => b.연도 === 2026 && b.단계_원문.includes("본예산")); if (budget2026) totalSum += budget2026.금액_백만원; }); // 2. 숫자 업데이트 (계산중 -> 실제 숫자) document.getElementById('total-budget').innerText = `${Math.round(totalSum).toLocaleString()} 백만원`; document.getElementById('total-projects').innerText = `${projects.length} 개`; // 3. 생키 차트 데이터 생성 (소관 -> 분야) const flowData = []; projects.forEach(p => { const budget2026 = p.예산.find(b => b.연도 === 2026 && b.단계_원문.includes("본예산")); if (budget2026 && budget2026.금액_백만원 > 0) { flowData.push([p.분류.소관, p.분류.분야.명, budget2026.금액_백만원]); } }); drawSankeyChart(flowData); }) .catch(error => { console.error("에러 발생:", error); document.getElementById('total-budget').innerText = "데이터 오류"; }); function drawSankeyChart(flowData) { google.charts.load('current', {'packages':['sankey']}); google.charts.setOnLoadCallback(() => { var data = new google.visualization.DataTable(); data.addColumn('string', 'From'); data.addColumn('string', 'To'); data.addColumn('number', 'Weight'); data.addRows(flowData); var chart = new google.visualization.Sankey(document.getElementById('sankey_basic')); chart.draw(data, { width: '100%', height: 500 }); }); }
