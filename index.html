// 기존의 fetch 부분을 아래와 같이 수정하세요. async function initDashboard() { try { const response = await fetch('./AI_재정사업_표준화_533.json'); if (!response.ok) throw new Error("JSON 파일을 찾을 수 없습니다."); // 1. 텍스트로 먼저 읽어옵니다. let textData = await response.text(); // 2. 중요: JSON 내의 NaN을 0으로 강제 치환합니다. // ": NaN" 형태나 ", NaN" 형태를 찾아 ": 0"으로 바꿉니다. textData = textData.replace(/:\s?NaN/g, ': 0'); // 3. 치환된 텍스트를 JSON 객체로 변환합니다. const data = JSON.parse(textData); const projects = data.projects; // --- 이후 로직은 동일하게 진행 --- let totalSum = 0; projects.forEach(p => { // 예산 데이터가 배열인지 확인 후 계산 if (p.예산 && Array.isArray(p.예산)) { const b2026 = p.예산.find(b => b.연도 === 2026 && b.단계_원문.includes("본예산")); if (b2026) totalSum += b2026.금액_백만원; } }); document.getElementById('total-projects').innerText = projects.length + " 개"; document.getElementById('total-budget').innerText = Math.round(totalSum).toLocaleString(); const flowMap = new Map(); projects.forEach(p => { if (p.예산 && Array.isArray(p.예산)) { const b2026 = p.예산.find(b => b.연도 === 2026 && b.단계_원문.includes("본예산")); if (b2026 && b2026.금액_백만원 > 0) { const from = p.분류.소관 || "미분류"; const to = p.분류.분야.명 || "미분류"; const key = `${from}→${to}`; flowMap.set(key, (flowMap.get(key) || 0) + b2026.금액_백만원); } } }); const flowData = Array.from(flowMap.entries()).map(([key, val]) => { const [from, to] = key.split('→'); return [from, to, val]; }); if (flowData.length === 0) { throw new Error("시각화할 수 있는 예산 데이터가 없습니다."); } google.charts.setOnLoadCallback(() => { const dataTable = new google.visualization.DataTable(); dataTable.addColumn('string', 'From'); dataTable.addColumn('string', 'To'); dataTable.addColumn('number', 'Weight'); dataTable.addRows(flowData); const options = { sankey: { node: { label: { fontSize: 12 }, interactivity: true, width: 15 }, link: { colorMode: 'gradient' } } }; const chart = new google.visualization.Sankey(document.getElementById('sankey_basic')); chart.draw(dataTable, options); }); } catch (error) { console.error(error); document.getElementById('sankey_basic').innerHTML = `
에러 발생: ${error.message}
`; } }
